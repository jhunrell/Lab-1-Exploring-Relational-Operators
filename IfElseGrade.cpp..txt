#include <iostream>
using namespace std;

int main() {
    int grade;

    // Ask for student's grade
    cout << "Enter grade: ";
    cin >> grade;

    // Two-way selection: pass or fail
    if (grade >= 75)
        cout << "Result: Pass" << endl;
    else
        cout << "Result: Fail" << endl;

    return 0;
}
