#include <iostream>
using namespace std;

int main() {
    int a = 5, b = 10, c = 5;

    cout << "a == b: " << (a == b) << endl;   // false → 0
    cout << "a != c: " << (a != c) << endl;   // false → 0
    cout << "a < b: " << (a < b) << endl;     // true → 1
    cout << "b > c: " << (b > c) << endl;     // true → 1
    cout << "(a + c) <= b: " << ((a + c) <= b) << endl; // true → 1

    cout << "'A' < 'a': " << ('A' < 'a') << endl; // true → 1
    return 0;
}
